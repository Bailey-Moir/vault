$$\begin{align*}
	\{a_n\} \text{ bounded above} &\iff \exists M : \forall n \in \Z^+, a_n < M \\
	\{a_n\} \text{ bounded below} &\iff \exists N : \forall n \in \Z^+,  N<a_n \\
	\{a_n\} \text{ bounded} &\iff \exists N,M : \forall n \in \Z^+, N<a_n<M
\end{align*}$$
